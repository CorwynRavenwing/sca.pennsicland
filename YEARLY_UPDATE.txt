# UPDATES TO DO PERIODICALLY, TO CLEAN UP THE DATABASE, BECAUSE WE CAN'T SEE
# NEW USERS LOGGING ON FOR SOME REASON:

	# GIVE EVERYBODY A REAL TIME-CREATED:
	update user_information SET time_created = last_update, last_update = last_update
	WHERE time_created = 0;

	# CLEAN UP TEMPORARY-ACCOUNT FIELD:
	update user_information SET temporary_account = 'T', last_update = last_update
	WHERE temporary_account IS NULL;
	update user_information SET temporary_account = 'T', last_update = last_update
	WHERE temporary_account = '';
	update user_information SET temporary_account = 'F', last_update = last_update
	WHERE user_name NOT LIKE "TEMP\_%";
	update user_information SET temporary_account = 'T', last_update = last_update
	WHERE user_name     LIKE "TEMP\_%";
	delete from user_information where temporary_account = 'T'
	and last_update < date_sub(now() , interval 180 DAY)
	and time_created < date_sub(now() , interval 180 DAY);
	# NB: shouldn't we do this much, much more often than 180 days?

	# CLEAN UP ACTIVE_ACCOUNT FIELD:
	update user_information SET active_account = 'F', last_update = last_update
	WHERE active_account IS NULL;
	update user_information SET active_account = 'F', last_update = last_update
	WHERE active_account = '';
	update user_information SET active_account = 'T', last_update = last_update
	WHERE user_id in (SELECT user_id FROM `land_groups` where user_id <> 0);
